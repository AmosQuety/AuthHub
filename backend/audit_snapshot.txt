    1: import geoip from "geoip-lite";
    2: import prisma from "../db/client.js";
    3: 
    4: interface RiskContext {
    5:     userId: string;
    6:     ipAddress: string;
    7:     userAgent: string;
    8: }
    9: 
   10: export class RiskEngine {
   11:     /**
   12:      * Calculates an anomaly score (0-100) for a login attempt.
   13:      * Higher score = higher risk.
   14:      */
   15:     static async calculateRiskScore({ userId, ipAddress, userAgent }: RiskContext): Promise<number> {
   16:         let score = 0;
   17: 
   18:         // Skip scoring for local development loopback IPs
   19:         if (ipAddress === "127.0.0.1" || ipAddress === "::1") {
   20:             return 0;
   21:         }
   22: 
   23:         // 1. Fetch recent successful sessions for baseline
   24:         const recentSessions = await prisma.session.findMany({
   25:             where: { userId },
   26:             orderBy: { createdAt: "desc" },
   27:             take: 5,
   28:         });
   29: 
   30:         if (recentSessions.length === 0) {
   31:             // First time login is inherently slightly higher risk, but shouldn't trigger block
   32:             return 10;
   33:         }
   34: 
   35:         // 2. IP Geolocation Anomaly Detection (+50 pts)
   36:         const currentGeo = geoip.lookup(ipAddress);
   37:         const currentCountry = currentGeo?.country || "UNKNOWN";
   38: 
   39:         const knownCountries = new Set<string>();
   40:         recentSessions.forEach(session => {
   41:             if (session.ipAddress && session.ipAddress !== "127.0.0.1" && session.ipAddress !== "::1") {
   42:                 const geo = geoip.lookup(session.ipAddress);
   43:                 if (geo?.country) knownCountries.add(geo.country);
   44:             }
   45:         });
   46: 
   47:         if (knownCountries.size > 0 && currentCountry !== "UNKNOWN" && !knownCountries.has(currentCountry)) {
   48:             score += 50; // Logging in from a completely new country!
   49:         }
   50: 
   51:         // 3. User-Agent / Device Anomaly Detection (+30 pts)
   52:         const knownUserAgents = new Set<string>();
   53:         recentSessions.forEach(session => {
   54:             if (session.deviceInfo) knownUserAgents.add(session.deviceInfo);
   55:         });
   56: 
   57:         if (knownUserAgents.size > 0 && !knownUserAgents.has(userAgent)) {
   58:             score += 30; // Logging in from a completely new device/browser
   59:         }
   60: 
   61:         // Ensure score stays within 0-100 bounds

   53:   }

  318: export const refresh = async (req: Request, res: Response, next: NextFunction): Promise<void> => {
  319:   try {
  320:     const { refreshToken: oldRefreshToken } = req.cookies;
  321: 
  322:     if (!oldRefreshToken) {
  323:       res.status(401).json({ error: "No refresh token provided" });
  324:       return;
  325:     }
  326: 
  327:     // 1. Verify token signature and expiry
  328:     let payload;
  329:     try {
  330:       payload = await verifyToken(oldRefreshToken);
  331:     } catch {
  332:       res.status(401).json({ error: "Invalid or expired refresh token" });
  333:       return;
  334:     }
  335: 
  336:     const userId = payload.sub;
  337:     const sessionId = payload.sid as string | undefined;
  338: 
  339:     if (!userId || !sessionId) {
  340:       res.status(401).json({ error: "Malformed refresh token" });
  341:       return;
  342:     }
  343: 
  344:     // 2. Find and validate the existing session
  345:     const oldSession = await prisma.session.findUnique({
  346:       where: { id: sessionId },
  347:     });
  348: 
  349:     if (!oldSession || oldSession.userId !== userId) {
  350:       // Token exists but session is dead/revoked server-side
  351:       res.status(401).json({ error: "Session revoked" });
  352:       return;
  353:     }
  354: 
  355:     if (oldSession.expiresAt < new Date()) {
  356:       res.status(401).json({ error: "Session expired" });
  357:       return;
  358:     }
  359: 
  360:     // Verify the hash matches (detects token theft/replay if the DB hash doesn't match the presented token)
  361:     const isValidHash = await verifyPassword(oldSession.refreshTokenHash, oldRefreshToken);
  362:     if (!isValidHash) {
  363:       // THEFT DETECTED: The token is structurally valid but the DB hash does not match.
  364:       // This means the token was already rotated and someone (an attacker) is trying to use the old, stolen one.
  365:       console.warn(`[SECURITY] Refresh token replay detected for user ${userId}. Revoking entire family.`);
  366: 
  367:       // Nuke ALL sessions for this user to halt the attacker
  368:       await prisma.session.deleteMany({ where: { userId } });
  369: 
  370:       AuditService.log({
  371:         userId,
  372:         action: "TOKEN_REFRESH",
  373:         status: "BLOCKED",
  374:         ipAddress: req.ip || "unknown",
  375:         deviceInfo: req.headers["user-agent"] || "unknown",
  376:         details: { reason: "Token Replay Detected - Family Revoked", sessionId }
  377:       });
  378: 
  379:       res.status(401).json({ error: "Invalid refresh token. Security violation detected." });
  380:       return;
  381:     }
  382: 
  383:     // 3. Token Rotation — delete old session, create new one
  384:     // We use deleteMany to avoid throwing if the session was already removed by a concurrent request
  385:     await prisma.session.deleteMany({ where: { id: sessionId } });
  386: 
  387:     const newSession = await prisma.session.create({
  388:       data: {
  389:         userId: userId,
  390:         refreshTokenHash: "pending",
  391:         expiresAt: new Date(Date.now() + 7 * 24 * 60 * 60 * 1000), // Extended 7 days
  392:         deviceInfo: oldSession.deviceInfo, // Inherit device context
  393:         ipAddress: req.ip || "unknown",   // Update to current IP
  394:       },
  395:     });
  396: 
  397:     // 4. Fetch the user to retrieve current roles
  398:     const user = await prisma.user.findUnique({
  399:       where: { id: userId },
  400:     });
  401: 
  402:     if (!user) {
  403:       res.status(401).json({ error: "User not found" });
  404:       return;
  405:     }
  406: 
  407:     // 5. Generate fresh tokens linked to the new session
  408:     const { accessToken, refreshToken: newRefreshToken } = await generateTokens(userId, newSession.id, [], user.roles, user.name);
  409: 
  410:     // 5. Hash and save new refresh token
  411:     const newRefreshTokenHash = await hashPassword(newRefreshToken);
  412:     await prisma.session.update({
  413:       where: { id: newSession.id },
  414:       data: { refreshTokenHash: newRefreshTokenHash },
  415:     });
  416: 
  417:     // 6. Set new cookie
  418:     res.cookie("refreshToken", newRefreshToken, {
  419:       httpOnly: true,
  420:       secure: process.env.NODE_ENV === "production",
  421:       sameSite: "strict",
  422:       maxAge: 7 * 24 * 60 * 60 * 1000,
  423:     });
  424: 
  425:     res.json({
  426:       accessToken,
  427:       message: "Token refreshed successfully",
  428:     });
  429: 
  430:   } catch (error) {
  431:     next(error);
  432:   }
  433: };
